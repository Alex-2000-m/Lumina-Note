const generateMockNotes = (count) => {
  return Array.from({ length: count }).map((_, i) => ({
    id: `mock-${Date.now()}-${i}-${Math.random().toString(36).substr(2, 9)}`,
    title: `${TITLES[Math.floor(Math.random() * TITLES.length)]} #${i + 1}`,

  }));
};

const FILTERS = [
  { id: 'all', label: '全部', icon: LayoutGrid },
  { id: 'note', label: '笔记', icon: FileText },
  { id: 'todo', label: '待办', icon: CheckSquare },
  { id: 'image', label: '图片', icon: ImageIcon },
];

const BATCH_SIZE = 50; // 激进策略：大幅增加单次加载数量，构建厚实的缓冲区

function useColumnCount() {
  const [columns, setColumns] = useState(1);
  useEffect(() => {
    const updateColumns = () => {
      const width = window.innerWidth;
      if (width >= 1280) setColumns(4);
      else if (width >= 1024) setColumns(3);
      else if (width >= 640) setColumns(2);
      else setColumns(1);
    };
    updateColumns();
    window.addEventListener('resize', updateColumns);
    return () => window.removeEventListener('resize', updateColumns);
  }, []);
  return columns;
}

export default function NoteFlowApp() {
  // --- State ---
  const [currentView, setCurrentView] = useState('dashboard');
  const [notes, setNotes] = useState(() => {
    try {
      const saved = localStorage.getItem('flowNotesPro');
      return saved ? JSON.parse(saved) : INITIAL_NOTES;
    } catch (e) {
      return INITIAL_NOTES;
    }
  });
  
  const [searchQuery, setSearchQuery] = useState('');
  const [activeFilter, setActiveFilter] = useState('all');
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [currentNote, setCurrentNote] = useState(null);
  const [darkMode, setDarkMode] = useState(false);
  const [isSidebarOpen, setIsSidebarOpen] = useState(true);
  const [storageError, setStorageError] = useState(false);

  // --- Infinite Scroll State ---
  const [displayCount, setDisplayCount] = useState(BATCH_SIZE);
  const [isLoadingMore, setIsLoadingMore] = useState(false);
  const observerTarget = useRef(null);
  const [isPending, startTransition] = useTransition();

  const columnCount = useColumnCount();

  // --- Effects ---
  useEffect(() => {
    try {
      localStorage.setItem('flowNotesPro', JSON.stringify(notes));
      setStorageError(false);
    } catch (e) {
      console.error("LocalStorage full", e);
      setStorageError(true);
    }
  }, [notes]);

  useEffect(() => {
    if (darkMode) document.documentElement.classList.add('dark');
    else document.documentElement.classList.remove('dark');
  }, [darkMode]);

  // --- Logic ---
  const allFilteredNotes = useMemo(() => {
    let result = notes;
    if (activeFilter !== 'all') {
      if (activeFilter === 'image') result = result.filter(n => n.coverImage);
      else result = result.filter(n => n.type === activeFilter);
    }
    if (searchQuery.trim()) {
      const lowerQuery = searchQuery.toLowerCase();
      result = result.filter(note => 
        note.title.toLowerCase().includes(lowerQuery) || 
        note.content.toLowerCase().includes(lowerQuery) ||
        note.tags.some(tag => tag.toLowerCase().includes(lowerQuery))
      );
    }
    return result.sort((a, b) => b.updatedAt - a.updatedAt);
  }, [notes, searchQuery, activeFilter]);

  const visibleNotes = useMemo(() => {
    return allFilteredNotes.slice(0, displayCount);
  }, [allFilteredNotes, displayCount]);

  // --- 核心优化：贪心算法 + 稳定高度计算 ---
  const noteColumns = useMemo(() => {
    const cols = Array.from({ length: columnCount }, () => []);
    const colHeights = new Array(columnCount).fill(0);

    visibleNotes.forEach((note) => {
      // 估算高度，作为分列的依据 (无需非常精确，只要相对比例对即可)
      const baseHeight = 140; 
      const imageHeight = note.coverImage ? 160 : 0;
      const contentHeight = Math.min(note.content.length, 150); // 简化计算
      const todoHeight = note.type === 'todo' ? 50 : 0;
      
      const estimatedHeight = baseHeight + imageHeight + contentHeight + todoHeight;

      // 贪心策略：找最矮的列
      const minHeight = Math.min(...colHeights);
      const minIndex = colHeights.indexOf(minHeight);

      cols[minIndex].push(note);
      colHeights[minIndex] += estimatedHeight;
    });
    
    return cols;
  }, [visibleNotes, columnCount]);

  useEffect(() => {
    setDisplayCount(BATCH_SIZE);
  }, [searchQuery, activeFilter]);

  useEffect(() => {
    const observer = new IntersectionObserver(
      entries => {
        if (entries[0].isIntersecting) {
          if (visibleNotes.length < allFilteredNotes.length) {
            setIsLoadingMore(true);
            startTransition(() => {
              // 激进策略：几乎移除延迟，只要不阻塞主线程立即加载
              setTimeout(() => {
                setDisplayCount(prev => prev + BATCH_SIZE);
                setIsLoadingMore(false);
              }, 0); 
            });
          }
        }
      },
      // 激进策略：提前 2500px (约 2-3 个屏幕高度) 开始预加载
      // 确保用户划到下面时内容早已准备好
      { threshold: 0.1, rootMargin: '2500px' } 
    );

    if (observerTarget.current) observer.observe(observerTarget.current);
    return () => {
      if (observerTarget.current) observer.unobserve(observerTarget.current);
    };
  }, [visibleNotes.length, allFilteredNotes.length]);

  // --- Handlers ---
  const handleSaveNote = (noteData) => {
    if (currentNote) {
      setNotes(prev => prev.map(n => n.id === currentNote.id ? { ...noteData, id: n.id, updatedAt: Date.now() } : n));
    } else {
      setNotes(prev => [{ ...noteData, id: crypto.randomUUID(), createdAt: Date.now(), updatedAt: Date.now() }, ...prev]);
    }
    setIsModalOpen(false);
    setCurrentNote(null);
  };

  const handleDeleteNote = useCallback((id) => {
    if (confirm('确认删除？')) setNotes(prev => prev.filter(n => n.id !== id));
  }, []);

  const handleGenerateData = () => {
    const count = 1000;
    setCurrentView('masonry');
    
    startTransition(() => {
      setTimeout(() => {
        const newNotes = generateMockNotes(count);
        setNotes(prev => {
          const combined = [...newNotes, ...prev];
          return combined.slice(0, 5000); 
        });
      }, 100);
    });
  };

  const handleClearData = () => {
    if(confirm('确定要清空所有数据吗？')) {
        setNotes([]);
        localStorage.removeItem('flowNotesPro');
        setStorageError(false);
    }
  };

  return (
    <div className={`flex h-screen overflow-hidden transition-colors duration-300 font-sans selection:bg-indigo-100 dark:selection:bg-indigo-900 ${darkMode ? 'bg-[#121212] text-slate-200' : 'bg-[#f7f7f9] text-slate-800'}`}>
      
      {/* Sidebar */}
      <aside 
        className={`${isSidebarOpen ? 'w-64 translate-x-0' : 'w-0 -translate-x-full opacity-0'} 
        flex-shrink-0 bg-white dark:bg-[#1a1a1a] border-r border-slate-200 dark:border-slate-800 transition-all duration-300 ease-in-out flex flex-col z-20 absolute md:relative h-full`}
      >
        <div className="p-6 flex items-center gap-3">
          <div className="w-8 h-8 bg-gradient-to-tr from-indigo-500 to-purple-600 rounded-lg flex items-center justify-center text-white shadow-lg shadow-indigo-200 dark:shadow-none">
            <BookOpen size={16} fill="currentColor" />
          </div>
          <span className="font-bold text-lg tracking-tight">ChronNote</span>
        </div>

        <nav className="flex-1 px-4 space-y-1 overflow-y-auto custom-scrollbar">
          <SidebarItem icon={Home} label="工作台" active={currentView === 'dashboard'} onClick={() => setCurrentView('dashboard')} />
          <SidebarItem icon={FileText} label="我的笔记" badge={notes.length} active={currentView === 'masonry'} onClick={() => setCurrentView('masonry')} />
          <SidebarItem icon={CheckSquare} label="待办事项" />
          
          <div className="pt-6 pb-2 px-2 text-xs font-semibold text-slate-400 dark:text-slate-500 uppercase tracking-wider">测试工具</div>
          <button onClick={handleGenerateData} className="w-full flex items-center gap-3 px-3 py-2 rounded-lg text-sm font-medium text-slate-600 dark:text-slate-400 hover:bg-slate-100 dark:hover:bg-slate-800 transition-colors mb-1">
            <Database size={18} className="text-emerald-500" />
            <span>生成 1K 测试数据</span>
          </button>
           <button onClick={handleClearData} className="w-full flex items-center gap-3 px-3 py-2 rounded-lg text-sm font-medium text-slate-600 dark:text-slate-400 hover:bg-slate-100 dark:hover:bg-slate-800 transition-colors mb-1">
            <Trash2 size={18} className="text-red-500" />
            <span>清空数据</span>
          </button>
          
          {storageError && (
            <div className="mt-4 p-3 bg-red-50 dark:bg-red-900/20 text-red-600 dark:text-red-400 text-xs rounded-lg border border-red-100 dark:border-red-800 flex items-start gap-2">
              <AlertTriangle size={14} className="flex-shrink-0 mt-0.5" />
              <span>本地存储已满。</span>
            </div>
          )}
        </nav>
        <div className="p-4 border-t border-slate-100 dark:border-slate-800 space-y-2">
          <button onClick={() => { setCurrentNote(null); setIsModalOpen(true); }} className="w-full py-2.5 bg-slate-900 dark:bg-slate-100 hover:bg-slate-800 dark:hover:bg-white text-white dark:text-black rounded-xl font-medium shadow-xl shadow-slate-200 dark:shadow-none transition-all flex items-center justify-center gap-2 group active:scale-95">
            <Plus size={18} className="group-hover:rotate-90 transition-transform" />
            <span>新建笔记</span>
          </button>
        </div>
      </aside>

      {/* Main Content */}
      <div className="flex-1 flex flex-col min-w-0 bg-[#f7f7f9] dark:bg-[#121212] relative transition-colors duration-300">
        <header className="h-16 flex items-center justify-between px-4 sm:px-8 border-b border-transparent bg-transparent sticky top-0 z-10">
          <div className="flex items-center gap-4 flex-1">
            <button onClick={() => setIsSidebarOpen(!isSidebarOpen)} className="p-2 -ml-2 text-slate-500 hover:bg-slate-100 dark:hover:bg-slate-800 rounded-lg transition-colors">
              <SidebarIcon size={20} />
            </button>
          </div>
          <div className="flex items-center gap-2">
             <button onClick={() => setDarkMode(!darkMode)} className="p-2 text-slate-500 hover:bg-slate-100 dark:hover:bg-slate-800 rounded-lg transition-colors">
              {darkMode ? <Sun size={18} /> : <Moon size={18} />}
            </button>
            <div className="w-8 h-8 rounded-full bg-gradient-to-r from-pink-500 to-rose-500 shadow-md ring-2 ring-white dark:ring-[#121212]"></div>
          </div>
        </header>

        {currentView === 'dashboard' ? (
          <DashboardView noteCount={notes.length} onViewChange={setCurrentView} onCreate={() => { setCurrentNote(null); setIsModalOpen(true); }} onGenerate={handleGenerateData} onClear={handleClearData} />
        ) : (
          <div className="flex-1 flex flex-col min-h-0 animate-in slide-in-from-right-4 duration-300">
            <div className="px-4 sm:px-8 py-4 flex flex-col sm:flex-row sm:items-center justify-between gap-4 border-b border-transparent">
              <div>
                <h2 className="text-2xl font-bold tracking-tight mb-1">我的笔记</h2>
                <div className="flex items-center gap-2 text-sm text-slate-500 dark:text-slate-400 h-5">
                  <span>已加载 {visibleNotes.length} / {allFilteredNotes.length}</span>
                  {(isLoadingMore || isPending) && <Loader2 size={12} className="animate-spin text-indigo-500" />}
                </div>
              </div>
              <div className="flex items-center gap-1 bg-slate-100 dark:bg-[#1e1e1e] p-1 rounded-xl self-start">
                {FILTERS.map(filter => (
                  <button key={filter.id} onClick={() => setActiveFilter(filter.id)} className={`flex items-center gap-2 px-3 py-1.5 rounded-lg text-xs font-medium transition-all ${activeFilter === filter.id ? 'bg-white dark:bg-slate-700 text-slate-900 dark:text-white shadow-sm' : 'text-slate-500 hover:text-slate-900 dark:hover:text-slate-300'}`}>
                    <filter.icon size={14} />
                    {filter.label}
                  </button>
                ))}
              </div>
            </div>

            {/* 修复：添加 overflow-x-hidden 防止横向滚动 */}
            <div className="flex-1 overflow-y-auto overflow-x-hidden px-4 sm:px-8 pb-10 pt-4 custom-scrollbar scroll-smooth">
               <div className="mb-6 relative max-w-md">
                 <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-slate-400" />
                 <input type="text" placeholder="实时筛选..." value={searchQuery} onChange={(e) => setSearchQuery(e.target.value)} className="w-full pl-10 pr-4 py-2 bg-white dark:bg-[#1e1e1e] border border-slate-200 dark:border-slate-800 rounded-lg text-sm focus:ring-2 focus:ring-indigo-500/20 outline-none transition-all placeholder:text-slate-400" />
               </div>

              {visibleNotes.length === 0 ? (
                <div className="h-64 flex flex-col items-center justify-center text-slate-400 border-2 border-dashed border-slate-200 dark:border-slate-800 rounded-2xl">
                  <Filter size={48} className="mb-4 opacity-50" />
                  <p>暂无相关内容</p>
                </div>
              ) : (
                <>
                  <div className="flex items-start gap-4 sm:gap-6 pb-4">
                    {noteColumns.map((columnNotes, colIndex) => (
                      /* 修复：添加 min-w-0 允许 Flex 子项收缩 */
                      <div key={colIndex} className="flex-1 flex flex-col gap-6 min-w-0">
                        {columnNotes.map(note => (
                          <ProfessionalCard 
                            key={note.id} 
                            note={note} 
                            onClick={() => { setCurrentNote(note); setIsModalOpen(true); }}
                            onDelete={handleDeleteNote}
                          />
                        ))}
                      </div>
                    ))}
                  </div>

                  <div ref={observerTarget} className="py-12 flex justify-center items-center w-full min-h-[50px]">
                    {visibleNotes.length < allFilteredNotes.length ? (
                      <div className="flex flex-col items-center gap-2 text-slate-400 opacity-0">Loading...</div>
                    ) : (
                      notes.length > 0 && <span className="text-xs text-slate-300 dark:text-slate-600">-- 到底了 --</span>
                    )}
                  </div>
                </>
              )}
            </div>
          </div>
        )}
      </div>

      {isModalOpen && <ProfessionalEditor note={currentNote} onClose={() => setIsModalOpen(false)} onSave={handleSaveNote} />}
    </div>
  );
}

function SidebarItem({ icon: Icon, label, active, badge, onClick }) {
  return (
    <button onClick={onClick} className={`w-full flex items-center justify-between px-3 py-2 rounded-lg text-sm font-medium transition-all mb-1 ${active ? 'bg-indigo-50 dark:bg-indigo-900/20 text-indigo-600 dark:text-indigo-400 shadow-sm' : 'text-slate-600 dark:text-slate-400 hover:bg-slate-100 dark:hover:bg-slate-800 hover:pl-4'}`}>
      <div className="flex items-center gap-3">
        <Icon size={18} className={active ? 'text-indigo-600 dark:text-indigo-400' : 'text-slate-400'} />
        {label}
      </div>
      {badge && <span className="bg-slate-200 dark:bg-slate-700 text-slate-600 dark:text-slate-300 text-[10px] px-1.5 py-0.5 rounded-md min-w-[20px] text-center">{badge}</span>}
    </button>
  );
}

function QuickAction({ icon: Icon, label, onClick, delay }) {
  return (
    <button onClick={onClick} className="flex flex-col items-center justify-center p-4 bg-white dark:bg-[#1e1e1e] rounded-xl border border-slate-200 dark:border-slate-800 shadow-sm hover:shadow-md hover:-translate-y-1 transition-all duration-300 group" style={{ animationDelay: `${delay}ms` }}>
      <div className="w-10 h-10 rounded-full bg-slate-50 dark:bg-slate-800 flex items-center justify-center text-slate-500 group-hover:bg-indigo-50 group-hover:text-indigo-600 dark:group-hover:bg-indigo-900/30 dark:group-hover:text-indigo-400 transition-colors mb-2">
        <Icon size={20} />
      </div>
      <span className="text-xs font-medium text-slate-600 dark:text-slate-300">{label}</span>
    </button>
  );
}

const ProfessionalCard = React.memo(({ note, onClick, onDelete }) => {
  const isTodo = note.type === 'todo';
  const previewLines = note.content.split('\n').slice(0, isTodo ? 4 : 3);
  
  return (
    <div 
      onClick={onClick} 
      style={{ contentVisibility: 'auto', containIntrinsicSize: '0 200px' }}
      className="group w-full bg-white dark:bg-[#1a1a1a] rounded-xl border border-slate-200/60 dark:border-slate-800 shadow-[0_2px_8px_-2px_rgba(0,0,0,0.05)] hover:shadow-[0_8px_20px_-4px_rgba(0,0,0,0.1)] hover:-translate-y-1 transition-transform duration-300 cursor-pointer overflow-hidden relative"
    >
      {note.coverImage && (
        <div className="h-40 w-full overflow-hidden relative bg-slate-100 dark:bg-slate-800">
          <img 
            src={note.coverImage} 
            alt="cover" 
            loading="lazy" 
            decoding="async" 
            className="w-full h-full object-cover transition-transform duration-500 group-hover:scale-105" 
          />
          <div className="absolute inset-0 bg-gradient-to-t from-black/20 to-transparent opacity-0 group-hover:opacity-100 transition-opacity" />
        </div>
      )}
      <div className="p-5">
        <h3 className="font-bold text-lg mb-2 text-slate-800 dark:text-slate-100 leading-snug break-words">{note.title || '无标题'}</h3>
        <div className="text-sm text-slate-500 dark:text-slate-400 leading-relaxed mb-4">
          {isTodo ? (
            <div className="space-y-1.5">
              {previewLines.map((line, i) => (
                <div key={i} className="flex items-center gap-2">
                  <div className={`w-3.5 h-3.5 rounded border flex-shrink-0 ${line.includes('[x]') ? 'bg-indigo-500 border-indigo-500' : 'border-slate-300 dark:border-slate-600'}`}>
                    {line.includes('[x]') && <CheckSquare size={12} className="text-white relative -top-px -left-px" />}
                  </div>
                  <span className={`truncate ${line.includes('[x]') ? 'line-through opacity-60' : ''}`}>{line.replace(/- \[[x ]\] /, '')}</span>
                </div>
              ))}
            </div>
          ) : (
            <p className="line-clamp-4 break-words">{note.content}</p>
          )}
        </div>
        <div className="flex items-center justify-between pt-3 border-t border-slate-50 dark:border-slate-800/50">
          <div className="flex gap-2 overflow-hidden flex-wrap h-6">
            {note.tags.map((tag, i) => <span key={i} className="text-[10px] uppercase font-semibold tracking-wider text-slate-400 bg-slate-50 dark:bg-slate-800/50 px-2 py-1 rounded-md">{tag}</span>)}
          </div>
          <span className="text-[10px] text-slate-300 dark:text-slate-600 shrink-0">{new Date(note.updatedAt).toLocaleDateString()}</span>
        </div>
        <div className="absolute top-3 right-3 opacity-0 group-hover:opacity-100 transition-opacity flex gap-1">
          <button onClick={(e) => { e.stopPropagation(); onDelete(note.id); }} className="p-1.5 bg-white dark:bg-slate-800 text-rose-500 rounded-lg shadow-sm border border-slate-100 dark:border-slate-700 hover:bg-rose-50"><Trash2 size={14} /></button>
        </div>
      </div>
    </div>
  );
}, (prev, next) => prev.note === next.note);

function ProfessionalEditor({ note, onClose, onSave }) {
  const [data, setData] = useState({
    title: note?.title || '',
    content: note?.content || '',
    coverImage: note?.coverImage || '',
    tags: note?.tags?.join(' ') || '',
    type: note?.type || 'note'
  });
  const handleChange = (key, val) => setData(prev => ({ ...prev, [key]: val }));
  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-slate-900/20 dark:bg-black/50 backdrop-blur-sm animate-in fade-in duration-200">
      <div className="w-full max-w-3xl bg-white dark:bg-[#1a1a1a] rounded-2xl shadow-2xl overflow-hidden flex flex-col max-h-[90vh] animate-in slide-in-from-bottom-4 duration-300">
        <div className="h-14 border-b border-slate-100 dark:border-slate-800 flex items-center justify-between px-6 bg-white dark:bg-[#1a1a1a] sticky top-0 z-10">
          <div className="flex items-center gap-1 text-slate-400">
            <Settings size={18} />
            <div className="w-px h-4 bg-slate-200 dark:bg-slate-700 mx-2" />
            <select value={data.type} onChange={(e) => handleChange('type', e.target.value)} className="bg-transparent text-sm font-medium text-slate-600 dark:text-slate-300 outline-none cursor-pointer">
              <option value="note">普通笔记</option>
              <option value="todo">待办清单</option>
            </select>
          </div>
          <div className="flex items-center gap-2">
            <button onClick={onClose} className="px-4 py-1.5 text-sm font-medium text-slate-500 hover:text-slate-700 dark:hover:text-slate-300 transition-colors">取消</button>
            <button onClick={() => onSave({ ...data, tags: data.tags.split(' ').filter(t => t) })} className="px-4 py-1.5 bg-indigo-600 hover:bg-indigo-700 text-white text-sm font-medium rounded-lg shadow-sm transition-colors">保存</button>
          </div>
        </div>
        <div className="flex-1 overflow-y-auto p-8 custom-scrollbar">
          <input type="text" placeholder="无标题" value={data.title} onChange={(e) => handleChange('title', e.target.value)} className="w-full text-4xl font-bold text-slate-800 dark:text-slate-100 placeholder:text-slate-300 dark:placeholder:text-slate-700 border-none outline-none bg-transparent mb-6" />
          <div className="flex flex-col gap-3 mb-8 text-sm">
             <div className="flex items-center gap-3 text-slate-400"><ImageIcon size={16} /><input type="text" placeholder="封面图片 URL (可选)" value={data.coverImage} onChange={(e) => handleChange('coverImage', e.target.value)} className="flex-1 bg-transparent border-none outline-none text-slate-600 dark:text-slate-300 placeholder:text-slate-300" /></div>
            <div className="flex items-center gap-3 text-slate-400"><Tag size={16} /><input type="text" placeholder="标签 (空格分隔)" value={data.tags} onChange={(e) => handleChange('tags', e.target.value)} className="flex-1 bg-transparent border-none outline-none text-slate-600 dark:text-slate-300 placeholder:text-slate-300" /></div>
          </div>
          <textarea placeholder="开始输入内容..." value={data.content} onChange={(e) => handleChange('content', e.target.value)} className="w-full h-[60vh] resize-none border-none outline-none text-lg text-slate-700 dark:text-slate-300 leading-relaxed bg-transparent placeholder:text-slate-300 font-serif" />
        </div>
      </div>
    </div>
  );
}
